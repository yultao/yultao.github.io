# 1343. Number of Sub-arrays of Size K and Average Greater than or Equal to Threshold

```java
class Solution {
    public int numOfSubarrays(int[] arr, int k, int threshold) {
        //sliding window
        int res = 0;
        int left = 0;
        int sum = 0;
        for(int i=0; i<arr.length; i++){
            //进
            sum += arr[i];
            if(i-left+1==k) {
                //算
                if(sum/k >= threshold) {
                    res++;
                }
                //出
                sum -= arr[left];
                left++;
            }
        }
        return res;
    }
}
```
