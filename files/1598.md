# 1598. Crawler Log Folder
[LeetCode](https://leetcode.com/problems/crawler-log-folder/description/)

```java
class Solution {
    public int minOperations(String[] logs) {
        //count
        int res = 0;
        for(String log : logs) {
            if("../".equals(log)) {
                if(res>0) {
                    res--;
                }
            } else if("./".equals(log)) {
                //do nothing
            } else {
                res++;
            }
        }
        return res;
    }

    public int minOperations2(String[] logs) {
        //stack
        Deque<String> stack = new ArrayDeque<>();
        for(String log : logs) {
            if("../".equals(log)) {
                if(!stack.isEmpty()) {
                    stack.pop();
                }
            } else if("./".equals(log)) {
                //do nothing
            } else {
                stack.push(log);
            }
        }
        return stack.size();
    }
}
```
