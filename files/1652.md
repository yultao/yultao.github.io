# 1652. Defuse the Bomb
[LeetCode](https://leetcode.com/problems/defuse-the-bomb/description/)

```java
class Solution {
    public int[] decrypt(int[] code, int k) {
        int n = code.length;
        int[] res = new int[n];
        if(k==0) return res;
        if(k>0) {
            //first sum
            res[0] = 0;
            for(int i = 1; i<=k; i++) {
                res[0] += code[i];
            }
            //or
            // int index = 1;
            // while(k-->0){
            //     res[0] += code[index];
            //     index++;
            // }

            //rest sum
            for(int i=1;i<n;i++){
                int ni = (i+k) % n;
                res[i] = res[i-1]-code[i] + code[ni];
            }
        } else {
            //last sum
            res[n-1] = 0;
            
            //use the second approach
            int index = n-2;
            int c = -k;
            while(c-->0){
                res[n-1] += code[index];
                index--;
            }
            

            //rest sum
            for(int i=n-2;i>=0;i--){
                int ni = (n+i+k) % n;
                res[i] = res[i+1]-code[i] + code[ni];
            }
        }

        //System.out.println("f: "+Arrays.toString(res));
        return res;
    }
}
```
