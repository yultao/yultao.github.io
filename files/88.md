# 88. Merge Sorted Array | M-54 | v

if we can return a new array
```java
class Solution {
    public int[] merge(int[] nums1, int m, int[] nums2, int n) {
        int[] res = new int[m+n];
        int i = 0;
        int j = 0;
        int c = 0;
        while(i<m && j<n) {
            if(nums1[i] < nums2[j]) {
                res[c++] = nums1[i++];
            } else {
                res[c++] = nums2[j++];
            }
        }
        while(i<m) {
            res[c++] = nums1[i++];
        }
        while(j<n) {
            res[c++] = nums2[j++];
        }
        return res;
    }
}
```

```java
class Solution {
    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int c = m+n-1;
        int i = m-1;
        int j = n-1;
        while(i>=0 && j>=0) {
            if(nums1[i] > nums2[j]) {
                nums1[c--] = nums1[i--];
            } else {
                nums1[c--] = nums2[j--];
            }
        }
        while(j>=0) {
            nums1[c--] = nums2[j--];
        }
    }
}
```

```java
class Solution {
    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int left = m-1;//nums1最后一个
        int right = n-1;//nums2最后一个
        int bottom = nums1.length-1;//合并后最后一个
        while(bottom>=0){

            if(left<0) {//只剩right
                nums1[bottom] = nums2[right];
                right--;
            } else if (right<0) {//只剩left
                nums1[bottom] = nums1[left];
                left--;

            } else if(nums1[left]<nums2[right]) {//选大的，right
                nums1[bottom] = nums2[right];
                right--;
            } else {
                nums1[bottom] = nums1[left];//选大的，left
                left--;
            }
            bottom--;
        }
    }
}
```
