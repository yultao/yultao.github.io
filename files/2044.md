# 2044. Count Number of Maximum Bitwise-OR Subsets
[LeetCode]()

```java
class Solution {
    private int res;
    public int countMaxOrSubsets(int[] nums) {
        int max = 0;
        for(int n : nums) {
            max |= n;
        }
        dfs(nums, 0, 0, max);
        return res;
    }
    //3, 2, 1, 5
    //3  215
    private void dfs(int[] nums, int value, int index, int max) {
        if(value==max) res++;
        for(int i=index;i<nums.length;i++) {
            dfs(nums, value|nums[i], i+1, max);
        }
    }
}
```
